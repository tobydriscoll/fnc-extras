function f = r3body(t,u)
x = u(1); y = u(2);
xdot = u(3); ydot = u(4);
mu = 0.012277471;
mustar = 1-mu;
re3 = ((x+mu)^2+y^2)^1.5;
rm3 = ((x-1+mu)^2+y^2)^1.5;
xdotdot = 2*ydot + x - (1-mu)*(x + mu)/re3 - mu*(x - 1 + mu)/rm3;
ydotdot = -2*xdot + y - (1-mu)*y/re3 - mu*y/rm3;
f = [xdot;ydot;xdotdot;ydotdot];
end
