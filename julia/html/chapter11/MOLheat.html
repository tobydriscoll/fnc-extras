tfinal = 0.05
using DifferentialEquations
ODE = (u,p,t) -> Dxx*u;  
IVP = ODEProblem(ODE,u0,(0,tfinal))

u = solve(IVP,DP5())  # select DP5 solver

using Plots
plot(x,u[:,1:50:end],label=["t=$t" for t in round.(u.t[1:50:end],digits=4)], 
    xaxis=("x"), yaxis=("u(x,t)"),
